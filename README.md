# README


# Rails Version
  Rails 5.0.0.1

# Ruby Version
  Ruby 2.3.1

# Database
  MySQL

# Run Gemfile
	bundle install
  
# Create Database
	rails db:create
  
# Run Migrations
	rails db:migrate
  
# Run Seed File
  Seed Data for storing the rooms category and rooms information to the database
	rails db:seed
 
# Start Server
	rails s
